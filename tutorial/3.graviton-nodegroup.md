$ KEY_PAIR = $(aws ec2 describe-key-pairs | grep -w "KeyName" | cut -d ':' -f 2 | sed 's/,//g' | sed 's/"//g')

$ eksctl create nodegroup \
  --cluster eks-grv-mig \
  --region ap-northeast-2 \
  --name ng-arm \
  --node-ami-family AmazonLinux2023	\
  --node-type c6g.2xlarge \
  --nodes 2 \
  --nodes-min 1 \
  --nodes-max 3 \
  --ssh-access \
  --ssh-public-key ${KEY_PAIR}
