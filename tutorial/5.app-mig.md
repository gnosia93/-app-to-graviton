### Graviton 노드 테인트 설정하기 ###
```
$ vi taint.sh
```
```
#!/bin/bash

NODE_LIST=$(kubectl get nodes -o wide | grep aarch64 | cut -d ' ' -f 1 | sed 'N; s/\n/;/')
IFS=';' read -ra my_array <<< "$NODE_LIST"

for i in "${my_array[@]}"
do
    kubectl taint node ${i} arch=arm:NoExecute	
done
```
```
sh taint.sh
```
