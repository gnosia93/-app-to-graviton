```
cat <<_EOF > prometheus.yml
global:
  scrape_interval:     15s  
  external_labels:
    monitor: 'monitor'

scrape_configs:
  - job_name: 'prometheus'
    scrape_interval: 5s
    static_configs:
      - targets: ['localhost:9090']
_EOF
```

```
./prometheus --config.file=prometheus.yml --storage.tsdb.path=~/data &
```
